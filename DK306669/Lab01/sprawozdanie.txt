git clone - za pomoca tej instrukcji klonuje repo

mkdir - tworzenie nowego folderu
touch - tworzenie nowego pliku

git checkout -b DK306669 - tworze nowy branch. aby na niego przejsc uzywam tej samej komendy ale bez -b
git add sprawozdanie.txt - dodaje pliki
git commit -m "1 commit - sprawozdanie" - zatwierdzam zmiany i nadaje jej opis dzieki -m

git push origin DK306669 - wypchniecie zmian
git pull - pobranie zmian z mastera

git merge master
git merge DK306669 - scalanie branchow
