Zadanie 2
Utworzylem nowego branach'a poleceniem git checkout -b Grupa01-GC306428_Lab05

Stworzylem nowa siec dla kontenera z Jenkinsem poleceniem "docker network create jenkins"

Uruchomilem obraz dockerowy docker:dind:
docker run \
  --name jenkins-docker \
  --rm \
  --detach \
  --privileged \
  --network jenkins \
  --network-alias docker \
  --env DOCKER_TLS_CERTDIR=/certs \
  --volume jenkins-docker-certs:/certs/client \
  --volume jenkins-data:/var/jenkins_home \
  --publish 2376:2376 \
  docker:dind \
  --storage-driver overlay2

Utowrzylem nowy kontener uzywajac Dockerfile poleceniem "docker build -t devops_jenkins ."

Uruchomilem kontener z Jenkinsem:
docker run \
  --name jenkins_devops_container \
  --rm \
  --detach \
  --network jenkins \
  --env DOCKER_HOST=tcp://docker:2376 \
  --env DOCKER_CERT_PATH=/certs/client \
  --env DOCKER_TLS_VERIFY=1 \
  --publish 8080:8080 \
  --publish 50000:50000 \
  --volume jenkins-data:/var/jenkins_home \
  --volume jenkins-docker-certs:/certs/client:ro \
  devops_jenkins
