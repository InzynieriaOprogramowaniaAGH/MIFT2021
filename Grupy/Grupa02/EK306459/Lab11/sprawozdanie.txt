2.1. instalacja minikube 

curl -LO https://storage.googleapis.com/minikube/releases/latest/minikube-linux-amd64
sudo install minikube-linux-amd64 /usr/local/bin/minikube

2.2. uruchomienie klastra

minikube start

2.3. pobranie kubectl


minikube kubectl -- get po -A
curl -LO "https://dl.k8s.io/release/$(curl -L -s https://dl.k8s.io/release/stable.txt)/bin/linux/amd64/kubectl"
sudo install -o root -g root -m 0755 kubectl /usr/local/bin/kubectl
kubectl version --client

3. uruchomienie dashboard

minikube dashboard

3.1. Funkcje dashboarda

	- zarzadzanie i monitorowanie zasobow
	- konfiguracja i uruchamianie klastrow
	- deployowac, monitorowac i skalowac aplikacje

4. uruchomienie komunikatora

kubectl run komunikator --image=YOUR_DOCKER_ID/YOUR_KOMUNIKATOR_IMG --port=9999 --labels app=komunikator

4.2. uruchomienie komunikatora na porcie 8080

kubectl create deployment hello-minikube1 --image=k8s.gcr.io/echoserver:1.4
kubectl expose deployment hello-minikube1 --type=LoadBalancer --port=8080
