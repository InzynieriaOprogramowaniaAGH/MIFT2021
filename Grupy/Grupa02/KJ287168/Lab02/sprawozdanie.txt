    ssh-keygen -t ed25519 -C "kjmj77@gmail.com"
    
    Hasła do klucza SSH (passphrase).

    Działający agent SSH do obsługi klucza SSH.
    eval "$(ssh-agent -s)"
    open ~/.ssh/config
    touch ~/.ssh/config
    vim ~/.ssh/config
    Host *
        AddKeysToAgent yes
        UseKeychain yes
        IdentityFile ~/.ssh/id_ed25519
    ssh-add -K ~/.ssh/id_ed25519
    
    Zarejestrowanie klucza po stronie serwera.
    pbcopy < ~/.ssh/id_ed25519.pub
