1. utworzono webhooka na githubie wedłgu poradnika na yt
2. pobrano ngrok by móc tunelowac localhost:8080 na zewnetrzny adres,
3. wklejono ten adres do konfiguracji webhooka,
4. ustawiono konfiguracje webhooka w pipeline
5. Rozszerzono Jenkinsfile i stage "Deploy"
stage('Deploy') {
			when {
              			expression {currentBuild.result == null || currentBuild.result == 'SUCCESS'}
            		}
            		steps {
                		echo 'Deploying....'
				sh 'docker build -t socketio_deploy -f Dockerfile_deploy .'
            		}
			post {
				always {
					emailext attachLog: true,
					body: "Jenkins Stage DEPLOY ended with status: ${currentBuild.currentResult}",
					to: 'szymonxlorenc@gmail.com', 
					subject: "Deploy: ${currentBuild.currentResult}"
				} 
			}
        	}
