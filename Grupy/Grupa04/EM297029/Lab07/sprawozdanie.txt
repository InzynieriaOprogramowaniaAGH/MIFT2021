4. Napisanie odpowiedniego Jenkinsfila i umieszczenie go w katalogu głównym nowego repozytorium lab07.
pipeline {
    agent {
             docker { image 'node:alpine'                                              #Wybrany agent odpowiada technologii jaka jest wykorzystywana w komunikatorze
                     }
             }
    stages {
        stage('Build'){ 
            steps{
		sh 'docker-compose up' 							#Tylko budowanie

            }
	}
	stage('Test'){
	    steps{
                sh 'code/wait.sh server:1234 -- echo READY && node code/client.js'	#testowanie
	    }
 	}
    }
    post {
    failure {
        mail to: 'emroz@student.agh.edu.pl',
             subject: "Failed Pipeline: ${currentBuild.fullDisplayName}",
             body: "Something is wrong with ${env.BUILD_URL}"
    }
    success {
            echo 'I succeeded :D'
    }
}
}
