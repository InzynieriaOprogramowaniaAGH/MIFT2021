1. Zainstalowano Ngrok, skonfigurowano webhook, dodano deploy key


2. Dodano nastepujace kroki do pliku Jenkinsfile: 
stage('Deploy') {
			steps {
				echo 'Deploy'
				sh 'docker build -t deltachat-deploy -f Dockerfile-deploy .'
			}
			post {
				success {
				    echo 'Success'
				    emailext attachLog: true,
					body: "Successful Deploy",
					to: 'zabludowskifilip@gmail.com',
					subject: "Successful Deploy"
				}
				failure {
				    echo 'Fail'
				    emailext attachLog: true,
				    	body: "Failed Deploy",
				       	to: 'zabludowskifilip@gmail.com',
					subject: "Failed Deploy"
			     	}
		    	}
		}
