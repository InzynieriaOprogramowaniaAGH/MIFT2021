3.Instalacja Dockera 
4.Założenie konta na DockerHUB. 
5.Stworzenie pliku Dockerfile:

RUN apt-get -y update
RUN apt-get -y upgrade 
RUN apt-get install -y build-essential 
RUN apt-get -y install git
